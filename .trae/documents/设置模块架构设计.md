## 🗃️ 三、状态管理设计

### 3.1 Zustand Store (settingsStore.ts)
```typescript
import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface PersonalSettings {
  theme: 'deep' | 'balanced' | 'clear';
  language: 'zh-CN' | 'en-US';
  notifications: {
    email: boolean;
    dingtalk: boolean;
    wechat: boolean;
  };
}

interface AlertRule {
  id: string;
  name: string;
  enabled: boolean;
  severity: 'info' | 'warning' | 'error' | 'critical';
  conditions: Record<string, unknown>;
}

interface DetectionSettings {
  models: Array<{
    id: string;
    name: string;
    enabled: boolean;
  }>;
  confidenceThreshold: number;
  debounceFrames: number;
}

interface SettingsState {
  personal: PersonalSettings;
  alerts: AlertRule[];
  detection: DetectionSettings;
  updatePersonal: (settings: Partial<PersonalSettings>) => void;
  updateAlerts: (rules: AlertRule[]) => void;
  updateDetection: (settings: Partial<DetectionSettings>) => void;
}

const defaultSettings: Omit<SettingsState, 'updatePersonal' | 'updateAlerts' | 'updateDetection'> = {
  personal: {
    theme: 'deep',
    language: 'zh-CN',
    notifications: {
      email: true,
      dingtalk: false,
      wechat: true,
    },
  },
  alerts: [],
  detection: {
    models: [],
    confidenceThreshold: 0.5,
    debounceFrames: 3,
  },
};

export const useSettingsStore = create<SettingsState>()(
  persist(
    (set) => ({
      ...defaultSettings,
      updatePersonal: (settings) =>
        set((state) => ({ personal: { ...state.personal, ...settings } })),
      updateAlerts: (rules) => set({ alerts: rules }),
      updateDetection: (settings) =>
        set((state) => ({ detection: { ...state.detection, ...settings } })),
    }),
    {
      name: 'xunjianbao-settings',
    }
  )
);
```

export type Theme = 'deep' | 'balanced' | 'clear';
export type Language = 'zh-CN' | 'en-US';

export interface PersonalSettings {
  theme: Theme;
  language: Language;
  notifications: {
    email: boolean;
    dingtalk: boolean;
    wechat: boolean;
  };
}

export interface AlertRule {
  id: string;
  name: string;
  enabled: boolean;
  severity: 'info' | 'warning' | 'error' | 'critical';
  conditions: {
    type: string;
    threshold: number;
    duration: number;
  };
  actions: Array<{
    type: 'notification' | 'webhook';
    channel: string;
  }>;
}

export interface DetectionModel {
  id: string;
  name: string;
  type: 'fire' | 'intrusion' | 'crack' | 'smoke' | 'vehicle';
  version: string;
  enabled: boolean;
}

export interface DetectionSettings {
  models: DetectionModel[];
  confidenceThreshold: number;
  debounceFrames: number;
  roi: Array<{
    x: number;
    y: number;
    width: number;
    height: number;
  }>;
}

export interface NotificationChannel {
  id: string;
  type: 'email' | 'dingtalk' | 'wechat' | 'webhook';
  name: string;
  enabled: boolean;
  config: Record<string, unknown>;
}

export interface SystemInfo {
  version: string;
  buildTime: string;
  uptime: number;
  resources: {
    cpu: number;
    memory: number;
    disk: number;
  };
}

## 🔌 五、API 设计 (api/v1/settings.ts)
```typescript
import { client } from '../client';
import type {
  PersonalSettings,
  AlertRule,
  DetectionSettings,
  NotificationChannel,
  SystemInfo,
} from '@/types/settings';

export const settingsApi = {
  // 个人设置
  getPersonalSettings: () => 
    client.get<PersonalSettings>('/api/v1/settings/personal'),
  
  updatePersonalSettings: (settings: Partial<PersonalSettings>) =>
    client.put('/api/v1/settings/personal', settings),

  // 告警规则
  getAlertRules: () =>
    client.get<AlertRule[]>('/api/v1/settings/alerts'),
  
  createAlertRule: (rule: Omit<AlertRule, 'id'>) =>
    client.post<AlertRule>('/api/v1/settings/alerts', rule),
  
  updateAlertRule: (id: string, rule: Partial<AlertRule>) =>
    client.put(`/api/v1/settings/alerts/${id}`, rule),
  
  deleteAlertRule: (id: string) =>
    client.delete(`/api/v1/settings/alerts/${id}`),

  // 检测设置
  getDetectionSettings: () =>
    client.get<DetectionSettings>('/api/v1/settings/detection'),
  
  updateDetectionSettings: (settings: Partial<DetectionSettings>) =>
    client.put('/api/v1/settings/detection', settings),

  // 通知渠道
  getNotificationChannels: () =>
    client.get<NotificationChannel[]>('/api/v1/settings/notifications'),
  
  updateNotificationChannel: (id: string, channel: Partial<NotificationChannel>) =>
    client.put(`/api/v1/settings/notifications/${id}`, channel),

  // 系统信息
  getSystemInfo: () =>
    client.get<SystemInfo>('/api/v1/system/info'),
};
```
